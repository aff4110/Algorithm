# [덱](https://www.acmicpc.net/problem/10866)

```
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());

        int[] list = new int[N*2];
        Arrays.fill(list, -1);

        StringBuilder sb = new StringBuilder();
        int idx1 = N+1, idx2 = N;
        for(int i=0; i<N; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());

            switch(st.nextToken()) {
                case "push_front":
                    list[--idx1] = Integer.parseInt(st.nextToken());
                    break;
                case "push_back":
                    list[++idx2] = Integer.parseInt(st.nextToken());
                    break;
                case "pop_front":
                    if((idx2-idx1) == -1) sb.append("-1\n");
                    else {
                        sb.append(list[idx1]).append("\n");
                        list[idx1++] = -1;
                    }
                    break;
                case "pop_back":
                    if((idx2-idx1) == -1) sb.append("-1\n");
                    else {
                        sb.append(list[idx2]).append("\n");
                        list[idx2--] = -1;
                    }
                    break;
                case "size":
                    sb.append(idx2-idx1+1).append("\n");
                    break;
                case "empty":
                    if((idx2-idx1) == -1) sb.append("1\n");
                    else sb.append("0\n");
                    break;
                case "front":
                    if((idx2-idx1) == -1) sb.append("-1\n");
                    else sb.append(list[idx1]).append("\n");
                    break;
                case "back":
                    if((idx2-idx1) == -1) sb.append("-1\n");
                    else sb.append(list[idx2]).append("\n");
                    break;
            }
        }
        System.out.print(sb);
    }
}
```
